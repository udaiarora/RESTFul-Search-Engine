Write Permanent SAS Data Set

//  JOB
//**********************************************************************
//*  JCL needed to create a new permanent SAS database.  The input data*
//*  is instream data (following the CARDS; statement).  This job will *
//*  create a permanent SAS database called TESTBASE (you can give it  *
//*  any name you want), with a SAS data set named TEST1 (this can also*
//*  be any name you want).  YOU MUST replace the following before this*
//*  job will run:                                                     *
//*                                                                    *
//*   REPLACE  IN                WITH                                  *
//*   -------  ----------------  --------------------------------------*
//*   xx       OUT DD statement  1st 2 letters of your userid (ie AG)  *
//*   yyy      OUT DD statement  Last 3 letters of userid (ie RBG)     *
//*   t        OUT DD statement  The primary track allocation* (ie 10) *
//*   tt       OUT DD statement  The secondary track allocation* (ie 2)*
//*                                                                    *
//*  You may OPTIONALLY replace either of the following:               *
//*                                                                    *
//*    REPLACE   IN                WITH                                *
//*    --------  ----------------  ---------------------------------   *
//*    TESTBASE  OUT DD statement  Any 8 character name you'd like**   *
//*    TEST1     DATA statement    Any 8 character name you'd like**   *
//*                                                                    *
//*        **Name must start with a letter, can have numbers after the *
//*            first character.                                        *
//*                                                                    *
//*  You must also replace the INPUT statement with one correctly      *
//*  describing your data.  And, you'll have to replace the data lines *
//*  after the CARDS statement with your own data.  You can also       *
//*  input your data from an external file as shown in the example     *
//*  program named READEXT.                                            *
//*                                                                    *
//*  You do not have to supply DCB statements when creating a permanent*
//*  SAS dataset, SAS will supply them automatically for you.          *
//**********************************************************************
//  EXEC  SAS
//OUT  DD  DISP=(NEW,CATLG),DSN=WYL.xx.yyy.TESTBASE, <==CHANGE AS NEEDED
//  UNIT=RESRCH,SPACE=(TRK,(t,tt))      <==CHANGE AS NEEDED

DATA OUT.TEST1;  /* You can replace TEST1 with any 8 character name */;
  INPUT A B C;  /* Replace the INPUT statement your own */;
CARDS;          /* Replace the data lines with your own */;
1 2 3
4 5 6
7 8 9
;

  /* Proc Contents will show you important information concerning
     your permanent SAS data set. */

PROC CONTENTS DATA=OUT._ALL_ MT=ALL HISTORY POSITION;
  TITLE 'PROC CONTENTS OF NEW PERMANENT SAS DATABASE';
//
