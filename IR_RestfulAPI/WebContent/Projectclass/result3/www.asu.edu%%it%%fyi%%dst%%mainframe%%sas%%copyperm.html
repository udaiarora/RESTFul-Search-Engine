PROC COPY

//  JOB
//**********************************************************************
//*  JCL needed to copy an existing permanent SAS database to a new    *
//*  permanent SAS database.  In this example, the existing database   *
//*  is named SASBASE1 and the new database being created is called    *
//*  SASBASE2 (you should replace these names with those that are      *
//*  meaningful to your needs).  By default, all SAS data sets in a    *
//*  database are copied over.  You may optionally SELECT or EXCLUDE   *
//*  certain data sets from being copied.  For more details, consult   *
//*  the SAS User's Guide: Basics.  This particular job is set up to   *
//*  copy all data sets in SASBASE1 into SASBASE2.  You can copy SAS   *
//*  databases from one account to another.  To do so, you must run    *
//*  this job from the account you want to copy the new database into. *
//*  You must have READ access to copy a database from anther account. *
//*  YOU MUST replace the following before this job can be run:        *
//*                                                                    *
//*   REPLACE  IN                WITH                                  *
//*   -------  ----------------  --------------------------------------*
//*   xx       IN  DD statement  1st 2 letters of a userid (ie AG)     *
//*   yyy      IN  DD statement  Last 3 letters of a userid (ie RBG)   *
//*   xx       OUT DD statement  1st 2 letters of a userid (ie AG)     *
//*   yyy      OUT DD statement  Last 3 letters of a userid (ie RBG)   *
//*                               (if copying from someone else's      *
//*                                account change xx and yyy in the IN *
//*                                DD statement to their userid)       *
//*   t        OUT DD statement  The primary track allocation* (ie 10) *
//*   tt       OUT DD statement  The secondary track allocation* (ie 2)*
//*                                                                    *
//*  You may OPTIONALLY replace either of the following:               *
//*                                                                    *
//*    REPLACE   IN                WITH                                *
//*    --------  ----------------  ---------------------------------   *
//*    SASBASE1  IN  DD statement  Existing SAS database's name        *
//*    SASBASE2  OUT DD statement  New SAS database's name**           *
//*                                                                    *
//*        **Name must start with a letter, can have numbers after the *
//*            first character, 8 characters or less.                  *
//*                                                                    *
//*  You do not have to supply DCB statements when creating a permanent*
//*  SAS dataset, SAS will supply them automatically for you.          *
//**********************************************************************
//  EXEC  SAS
//IN   DD  DSN=WYL.xx.yyy.SASBASE1,DISP=SHR
//OUT  DD  DISP=(NEW,CATLG),DSN=WYL.xx.yyy.SASBASE2, <==CHANGE AS NEEDED
//  UNIT=RESRCH,SPACE=(TRK,(t,tt))      <==CHANGE AS NEEDED

PROC COPY   INDD=IN   OUTDD=OUT  MT=ALL;

  /* Proc Contents will show you important information concerning
     your permanent SAS data set. */

PROC CONTENTS DATA=OUT._ALL_ MT=ALL HISTORY POSITION;
  TITLE 'PROC CONTENTS OF NEW PERMANENT SAS DATABASE';
//
