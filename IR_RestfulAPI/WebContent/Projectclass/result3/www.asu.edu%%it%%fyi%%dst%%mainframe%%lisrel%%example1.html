Example 1

// JOB
// EXEC SPSS
title "FROM 'LISREL 7: A Guide to the Program and Applications'"

matrix data  /* Note that SPSS variable names do not allow hyphens
  variables = STATUS ROWTYPE_ READ_GR5 WRIT_GR5
                              READ_GR7 WRIT_GR7
  / SPL=STATUS / FORMAT=LIST LOWER DIAGONAL

begin data
1 COV      281.349
1 COV      184.219 182.821
1 COV      216.739 171.699 283.289
1 COV      198.376 153.201 208.837 246.069
1 N        373     373     373     373
2 COV      174.485
2 COV      134.468 161.869
2 COV      129.840 118.836 228.449
2 COV      102.194  97.767 136.058 180.460
2 N        249     249     249     249
end data

list

lisrel
   /"EXAMPLE 9.1A: Hypothesis A (Boys Academic)"
   /"Testing Equality of Factor Structures.    "
   /DA NI=4
   /MO NX=4 NK=4 LX=ID TD=ZE
   /OU
   /"EXAMPLE 9.1A: Hypothesis A (Boys Non-Academic)"
   /DA
   /MO PH=IN
   /OU

lisrel
   /"EXAMPLE 9.1B: Hypothesis B (Boys Academic)"
   /"Testing Equality of Factor Structures.    "
   /DA NI=4 NG=2
   /MO NX=4 NK=2
   /FR LX 2 1 LX 4 2
   /VA 1 LX 1 1 LX 3 2
   /OU
   /"EXAMPLE 9.1B: Hypothesis A (Boys Non-Academic)"
   /DA
   /MO LX=PS
   /OU
finish
