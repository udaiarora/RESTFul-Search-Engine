PROC CONTENTS

//  JOB
//**********************************************************************
//*  JCL needed to run PROC CONTENTS on a permanent SAS data file. PROC*
//*  CONTENTS prints out a description of the information contained in *
//*  the permanent SAS data file indicated in the DD statement.  You do*
//*  not need to know the names of the SAS data sets you have in a     *
//*  database to do PROC CONTENTS. The option _ALL_ takes care of this.*
//*  Info on PROC CONTENTS can be found in the SAS Users Guide: Basics.*
//*  Change the DD statement as needed for your own dataset.  This par-*
//*  ticular example gives the contents of the SAS/GRAPH U.S. Counties *
//*  graphics database documented in the SAS/GRAPH User's Guide.       *
//**********************************************************************
//  EXEC  SAS
//PERM DD DSN=WYL.PD.ACD.SASGRAPH.COUNTIES,DISP=SHR <==Change as needed

PROC CONTENTS DATA=PERM._ALL_ MT=ALL HISTORY POSITION;
  TITLE 'PROC CONTENTS OF THE COUNTIES DATASET';  /*Change title as
                                                    needed */;
//
