Create the data set. REFLIB.SWIRL is generated data that produces a symmetric contour pattern, which is useful for illustrating the PATTERN option. 
data reflib.swirl;
   do x= -5 to 5 by 0.25;
      do y= -5 to 5 by 0.25;
         if x+y=0 then z=0;
         else z=(x*y)*((x*x-y*y)/(x*x+y*y));
         output;
      end;
   end;
run;
