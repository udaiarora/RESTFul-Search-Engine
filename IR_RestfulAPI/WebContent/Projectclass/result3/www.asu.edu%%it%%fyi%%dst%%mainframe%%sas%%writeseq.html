Write Sequential Data Set

//  JOB
//**********************************************************************
//*  JCL needed to create a sequential data file from SAS. Input is    *
//*  instream data (following the CARDS; statement).  This job will    *
//*  create a permanent sequential data set called TESTOUT (you can    *
//*  name it any name you want).  YOU MUST replace the following       *
//*  before this job will run:                                         *
//*                                                                    *
//*    REPLACE  IN                WITH                                 *
//*    -------  ----------------  -------------------------------------*
//*    xx       OUT DD statement  1st 2 letters of your userid (ie AG) *
//*    yyy      OUT DD statement  Last 3 letters of userid (ie RBG)    *
//*    t        OUT DD statement  The primary track allocation (ie 5)  *
//*    tt       OUT DD statement  The secondary track allocation (ie 2)*
//*                                                                    *
//*  You may OPTIONALLY replace the following:                         *
//*                                                                    *
//*    REPLACE   IN                 WITH                               *
//*    --------  -----------------  ---------------------------------  *
//*    TESTOUT   OUT DD statement   Any 8 character name you'd like*   *
//*    80        LRECL statement    A valid record length if not 80    *
//*    7440      BLKSIZE statement  Any even multiple of LRECL less    *
//*                                 than or equal to the number 7476   *
//*                                                                    *
//*        *Name must start with a letter, can have numbers after the  *
//*         first character.                                           *
//*                                                                    *
//*  You will also have to replace the INPUT and PUT statements with   *
//*  ones correctly describing your data.  Also, you'll have to re-    *
//*  place the data lines after the CARDS statement with your own data.*
//*  Your input data can also come from an external file as shown in   *
//*  the example READEXT.                                              *
//**********************************************************************
//  EXEC  SAS
//OUT  DD  DISP=(NEW,CATLG),DSN=WYL.xx.yyy.TESTOUT, <==CHANGE AS NEEDED
//  UNIT=RESRCH,SPACE=(TRK,(t,tt),RLSE),     <==CHANGE AS NEEDED
//  DCB=(RECFM=FB,LRECL=80,BLKSIZE=7440)     <==CHANGE IF NEEDED

DATA IN;
  INPUT A B C;  /* Replace the INPUT statement your own */;
CARDS;          /* Replace the data lines with your own */;
1 2 3
4 5 6
7 8 9
;

DATA _NULL_;
  SET IN;
  FILE OUT;
  PUT A 1 B 3 C 5;  /* Replace PUT statement with one relating to
                       your data.  See SAS Basics manual for more
                       info on the PUT statement. */
//
